doubleMe x = x*x
doubleYou x y = doubleMe x + doubleMe y
doubleSmallNumber x = succ (if x>100 then x else x*2) 
boom xs = [if x<10 then "boom" else "bang" | x<-xs , odd x ]
fact :: (Integral a) => a -> a
fact 0 = 1
fact a = a * fact (a-1)
length' :: (Num b) => [a] -> b
length' [] = 0
length' (_:xs) = 1 + length' xs
sum' :: (Num a) => [a] -> a
sum' [] = 0
sum' (x:xs) = x + sum' xs
describeList :: [a] -> String
describeList xs = "The list is " ++ case xs of [] -> "empty."
                                               [x] -> "a singleton list."   
                                               xs -> "a longer list."

describeList2 :: [a] -> String
describeList2 xs = "The list is " ++ what xs 
    where what [] = "empty."
          what [x] = "a singleton list."
          what xs = "a longer list."

data Color = Red | Blue | Green deriving (Eq, Ord, Enum, Read,Show)

colors :: [Color]
colors = [Red, Blue, Green]

color = Red

nextSeason :: Color -> Color
nextSeason Red = Blue
nextSeason Blue = Green
nextSeason Green = Red

func :: Color -> String
func color 
    | color == Red = "Red"
    | color == Blue = "Blue"
    | color == Green = "Green"
